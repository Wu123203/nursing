```java
    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .csrf().disable()
            .sessionManagement().sessionCreationPolicy(SessionCreationPolicy.STATELESS)
```

### 1.3 验证码优化

**问题描述：**
- 验证码无过期时间
- 验证码复杂度较低
- 存在暴力破解风险

**优化方案：**
```java
// 验证码服务优化
@Service
public class CaptchaService {
    
    // 验证码有效期：5分钟
    private static final long CAPTCHA_EXPIRE_TIME = 5 * 60 * 1000;
    
    // 存储验证码和生成时间
    private final Map<String, CaptchaInfo> captchaCache = new ConcurrentHashMap<>();
    
    public void storeCaptcha(String sessionId, String captcha) {
        captchaCache.put(sessionId, new CaptchaInfo(captcha, System.currentTimeMillis()));
    }
    
    public boolean validateCaptcha(String sessionId, String input) {
        CaptchaInfo info = captchaCache.get(sessionId);
        if (info == null) {
            return false;
        }
        
        // 检查是否过期
        if (System.currentTimeMillis() - info.getTimestamp() > CAPTCHA_EXPIRE_TIME) {
            captchaCache.remove(sessionId);
            return false;
        }
        
        // 验证后立即删除
        captchaCache.remove(sessionId);
        return info.getCaptcha().equalsIgnoreCase(input);
    }
}
```

**实施步骤：**
1. 添加验证码过期机制
2. 增加图形验证码复杂度（干扰线、噪点）
3. 限制验证码尝试次数
4. 考虑使用 Redis 存储验证码

**预期收益：**
- 防止暴力破解
- 提高系统安全性

```java
        logger.error("系统异常: ", e);
```

```java
// Redis 配置类
@Configuration
@EnableCaching
public class RedisConfig {

    @Bean
    public RedisTemplate<String, Object> redisTemplate(RedisConnectionFactory factory) {
        RedisTemplate<String, Object> template = new RedisTemplate<>();
        template.setConnectionFactory(factory);
        // Key 使用 String 序列化
        template.setKeySerializer(new StringRedisSerializer());
        template.setHashKeySerializer(new StringRedisSerializer());
        // Value 使用 JSON 序列化
        template.setValueSerializer(new GenericJackson2JsonRedisSerializer());
        template.setHashValueSerializer(new GenericJackson2JsonRedisSerializer());
        template.afterPropertiesSet();
        return template;
    }

    @Bean
    public CacheManager cacheManager(RedisConnectionFactory factory) {
        RedisCacheConfiguration config = RedisCacheConfiguration.defaultCacheConfig()
                .entryTtl(Duration.ofHours(1))  // 默认缓存1小时
                .serializeKeysWith(RedisSerializationContext.SerializationPair
                        .fromSerializer(new StringRedisSerializer()))
                .serializeValuesWith(RedisSerializationContext.SerializationPair
                        .fromSerializer(new GenericJackson2JsonRedisSerializer()));

        return RedisCacheManager.builder(factory)
                .cacheDefaults(config)
                .build();
    }
}

// Redis 工具类，封装常用操作
@Component
public class RedisUtil {

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    // 设置值
    public void set(String key, Object value) {
        redisTemplate.opsForValue().set(key, value);
    }

    // 设置值并指定过期时间（秒）
    public void set(String key, Object value, long timeout) {
        redisTemplate.opsForValue().set(key, value, timeout, TimeUnit.SECONDS);
    }

    // 获取值
    public Object get(String key) {
        return redisTemplate.opsForValue().get(key);
    }

    // 删除键
    public void del(String key) {
        redisTemplate.delete(key);
    }

    // 设置过期时间
    public boolean expire(String key, long timeout) {
        return Boolean.TRUE.equals(redisTemplate.expire(key, timeout, TimeUnit.SECONDS));
    }

    // 获取过期时间
    public long getExpire(String key) {
        Long expire = redisTemplate.getExpire(key, TimeUnit.SECONDS);
        return expire != null ? expire : -1;
    }

    // 判断键是否存在
    public boolean hasKey(String key) {
        return Boolean.TRUE.equals(redisTemplate.hasKey(key));
    }

    // 自增
    public long incr(String key, long delta) {
        Long result = redisTemplate.opsForValue().increment(key, delta);
        return result != null ? result : 0;
    }

    // 自减
    public long decr(String key, long delta) {
        Long result = redisTemplate.opsForValue().decrement(key, delta);
        return result != null ? result : 0;
    }
}
```

#### 2.5.3 热点数据缓存示例 — 新闻资讯
```java
@Service
public class NewsServiceImpl implements NewsService {

    @Autowired
    private NewsMapper newsMapper;

    @Autowired
    private RedisUtil redisUtil;

    // 获取单条新闻详情
    public News getNewsById(Integer id) {
        String key = RedisConstants.NEWS_KEY + id;

        // 先查缓存
        Object obj = redisUtil.get(key);
        if (obj != null) {
            return (News) obj;
        }

        // 缓存未命中，查数据库
        News news = newsMapper.selectById(id);
        if (news != null) {
            // 写入缓存
            redisUtil.set(key, news, RedisConstants.NEWS_TTL);
        }
        return news;
    }

    // 获取新闻列表
    public List<News> getNewsList(Integer category, Integer pageNum, Integer pageSize) {
        String key = RedisConstants.NEWS_LIST_KEY + category + ":" + pageNum + ":" + pageSize;

        Object obj = redisUtil.get(key);
        if (obj != null) {
            return (List<News>) obj;
        }

        PageHelper.startPage(pageNum, pageSize);
        List<News> list = newsMapper.selectByCategory(category);

        // 注意：列表缓存更新策略，当有新闻新增/修改时需删除缓存
        if (list != null && !list.isEmpty()) {
            redisUtil.set(key, list, RedisConstants.NEWS_TTL);
        }
        return list;
    }

    // 新增新闻 — 删除相关缓存
    public boolean addNews(News news) {
        int result = newsMapper.insert(news);
        if (result > 0) {
            // 删除所有新闻列表缓存（实际需要遍历 key，推荐用模糊匹配）
            Set<String> keys = redisTemplate.keys(RedisConstants.NEWS_LIST_KEY + "*");
            if (keys != null && !keys.isEmpty()) {
                redisTemplate.delete(keys);
            }
        }
        return result > 0;
    }

    // 修改新闻 — 删除单条和列表缓存
    public boolean updateNews(News news) {
        int result = newsMapper.updateById(news);
        if (result > 0) {
            // 删除单条缓存
            redisUtil.del(RedisConstants.NEWS_KEY + news.getId());
            // 删除列表缓存
            Set<String> keys = redisTemplate.keys(RedisConstants.NEWS_LIST_KEY + "*");
            if (keys != null && !keys.isEmpty()) {
                redisTemplate.delete(keys);
            }
        }
        return result > 0;
    }
}
```

```java
// 修改验证码生成
@GetMapping("/captcha/{l}/{w}/{h}")
public JSONReturn getCaptcha(@PathVariable Integer l,
                             @PathVariable Integer w,
                             @PathVariable Integer h) {
    CodeUtil codeUtil = new CodeUtil(l, w, h);
    String code = codeUtil.generatorVCode();
    BufferedImage image = codeUtil.generatorRotateVCodeImage(code, true);

    // 生成验证码的唯一 key，返回给前端
    String captchaKey = UUID.randomUUID().toString().replace("-", "");
    String redisKey = RedisConstants.CAPTCHA_KEY + captchaKey;

    // 将验证码存入 Redis，设置过期时间
    redisUtil.set(redisKey, code, RedisConstants.CAPTCHA_TTL);

    FastByteArrayOutputStream os = new FastByteArrayOutputStream();
    ImageIO.write(image, "gif", os);
    String base64Img = Base64Utils.encodeToString(os.toByteArray());

    // 返回验证码图片和 key
    Map<String, String> result = new HashMap<>();
    result.put("image", base64Img);
    result.put("captchaKey", captchaKey);
    return JSONReturn.success(result);
}

// 修改登录验证
public JSONReturn login(LoginDto loginDto) {
    // 从 Redis 中获取验证码
    String redisKey = RedisConstants.CAPTCHA_KEY + loginDto.getCaptchaKey();
    Object captchaObj = redisUtil.get(redisKey);

    if (captchaObj == null) {

    if (!realCaptcha.equalsIgnoreCase(loginDto.getCaptcha())) {
        return JSONReturn.failed("验证码错误");
    }

    // 验证通过后删除验证码（一次性使用）
    redisUtil.del(redisKey);

    // 继续登录逻辑...
}
```

#### 3.5.1 接口限流保护（防止刷接口/暴力破解）

```java
// 限流注解
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
public @interface RateLimit {
    // 每分钟允许的请求次数
    int limit() default 10;
    // 限流 key（可结合 IP、用户 ID）
    String key() default "";
}

// 限流拦截器
@Component
public class RateLimitInterceptor implements HandlerInterceptor {

    @Autowired
    private RedisUtil redisUtil;

    @Override
    public boolean preHandle(HttpServletRequest request,
                             HttpServletResponse response,
                             Object handler) throws Exception {
        if (!(handler instanceof HandlerMethod)) {
            return true;
        }

        HandlerMethod method = (HandlerMethod) handler;
        RateLimit rateLimit = method.getMethodAnnotation(RateLimit.class);

        if (rateLimit == null) {
            return true;
        }

        // 拼接限流 key：rate_limit:接口路径:客户端IP
        String ip = request.getRemoteAddr();
        String key = RedisConstants.RATE_LIMIT_KEY
                   + method.getMethod().getName() + ":" + ip;

        // 从 Redis 获取当前计数
        Object obj = redisUtil.get(key);
        long count = 0;

        if (obj == null) {
            // 首次访问，设置初始值和过期时间
            redisUtil.set(key, 1, RedisConstants.RATE_LIMIT_TTL);
            count = 1;
        } else {
            count = redisUtil.incr(key, 1);
        }

        if (count > rateLimit.limit()) {
            response.setContentType("application/json;charset=UTF-8");
            response.getWriter().write(
                JSONReturn.failed("请求过于频繁，请稍后再试").toJsonString()
            );
            return false;
        }

        return true;
    }
}
```

#### 3.5.2 分布式锁（防止重复提交/并发抢占）

使用 Redis 的 SETNX（SET if Not eXists）特性实现分布式锁。

```java
// 分布式锁工具类
@Component
public class DistributedLock {

    @Autowired
    private StringRedisTemplate redisTemplate;

    /**
     * 尝试获取锁
     * @param lockKey  锁的 key
     * @param requestId  请求标识（用于释放锁时校验）
     * @param expireTime 过期时间（秒）
     * @return 是否获取成功
     */
    public boolean tryLock(String lockKey, String requestId, long expireTime) {
        // SET key value NX EX seconds
        // NX = key 不存在才设置成功
        // EX = 设置过期时间
        Boolean result = redisTemplate.opsForValue()
            .setIfAbsent(lockKey, requestId, expireTime, TimeUnit.SECONDS);
        return Boolean.TRUE.equals(result);
    }

    /**
     * 释放锁（Lua 脚本保证原子性：校验 + 删除）
     */
    public boolean releaseLock(String lockKey, String requestId) {
        String script =
            "if redis.call('get', KEYS[1]) == ARGV[1] then" +
            "    return redis.call('del', KEYS[1])" +
            "else" +
            "    return 0" +
            "end";

        DefaultRedisScript<Long> redisScript = new DefaultRedisScript<>();
        redisScript.setScriptText(script);
        redisScript.setResultType(Long.class);

        Long result = redisTemplate.execute(redisScript,
                Collections.singletonList(lockKey), requestId);
        return result != null && result == 1;
    }
}

// 使用示例 — 用户提交预约，防止重复点击
@PostMapping("/appointment/submit")
public JSONReturn submitAppointment(@RequestBody AppointmentDto dto) {
    // 基于用户 ID 生成锁 key
    String lockKey = RedisConstants.LOCK_KEY + "appointment:" + dto.getUserId();
    String requestId = UUID.randomUUID().toString();

    // 尝试获取锁，30 秒自动过期
    boolean locked = distributedLock.tryLock(
        lockKey, requestId, RedisConstants.LOCK_TTL
    );

    if (!locked) {
        return JSONReturn.failed("您的操作过于频繁，请稍后再试");
    }

    try {
        // 执行业务逻辑
        return appointmentService.submit(dto);
    } finally {
        // 释放锁
        distributedLock.releaseLock(lockKey, requestId);
    }
}

// 使用示例 — 防止重复提交订单
public boolean createOrder(OrderDto dto) {
    String lockKey = RedisConstants.LOCK_KEY + "order:" + dto.getUserId();
    String requestId = UUID.randomUUID().toString();

    // 尝试获取锁，设置 10 秒过期
    if (!distributedLock.tryLock(lockKey, requestId, 10)) {
        throw new BizException("订单正在处理中，请勿重复提交");
    }

    try {
        // 检查用户是否有未完成订单
        Order pending = orderMapper.selectPendingByUserId(dto.getUserId());
        if (pending != null) {
            throw new BizException("您已有未完成的订单");
        }

        // 创建订单
        return orderMapper.insert(dto) > 0;
    } finally {
        distributedLock.releaseLock(lockKey, requestId);
    }
}
```

#### 3.5.3 页面浏览量/点赞数统计（Redis 增量计数）

高频计数操作直接在 Redis 中完成，避免频繁写入数据库。

```java
@Service
public class ViewCountService {

    @Autowired
    private RedisUtil redisUtil;

    @Autowired
    private NewsMapper newsMapper;

    /**
     * 增加新闻浏览量（每次用户访问新闻详情时调用）
     */
    public void incrementViewCount(Integer newsId) {
        String key = RedisConstants.PAGE_VIEW_KEY + newsId;
        // Redis 自增，原子操作
        redisUtil.incr(key, 1);
    }

    /**
     * 增加新闻点赞数
     */
    public boolean incrementLikeCount(Integer newsId, Integer userId) {
        // 先检查用户是否已点赞（用 Set 存储点赞用户 ID）
        String likeUserKey = RedisConstants.LIKE_KEY + newsId + ":users";
        Boolean member = redisTemplate.opsForSet()
            .isMember(likeUserKey, userId.toString());

        if (Boolean.TRUE.equals(member)) {
            return false;  // 已点赞
        }

        // 添加用户到点赞集合
        redisTemplate.opsForSet().add(likeUserKey, userId.toString());
        // 点赞数 +1
        String likeCountKey = RedisConstants.LIKE_KEY + newsId + ":count";
        redisUtil.incr(likeCountKey, 1);
        return true;
    }

    /**
     * 获取浏览量
     */
    public long getViewCount(Integer newsId) {
        String key = RedisConstants.PAGE_VIEW_KEY + newsId;
        Object obj = redisUtil.get(key);
        if (obj == null) {
            // 缓存未命中，从数据库查询
            News news = newsMapper.selectById(newsId);
            long count = news != null ? news.getViewCount() : 0;
            // 同步到 Redis
            if (count > 0) {
                redisUtil.set(key, count);
            }
            return count;
        }
        return Long.parseLong(obj.toString());
    }

    /**
     * 定时任务：将 Redis 中的计数同步到数据库（如每 10 分钟）
     */
    @Scheduled(cron = "0 */10 * * * ?")
    public void syncCountToDatabase() {
        // 遍历所有 page_view 前缀的 key
        Set<String> keys = redisTemplate.keys(RedisConstants.PAGE_VIEW_KEY + "*");
        if (keys == null || keys.isEmpty()) {
            return;
        }

        for (String key : keys) {
            Object value = redisUtil.get(key);
            if (value == null) continue;

            Integer newsId = Integer.parseInt(
                key.substring(RedisConstants.PAGE_VIEW_KEY.length())
            );
            long count = Long.parseLong(value.toString());

            // 批量更新到数据库
            newsMapper.updateViewCount(newsId, count);
        }
    }
}
```

#### 3.5.4 JWT Token 黑名单管理（登出主动失效）

由于 JWT 本身是无状态的，默认无法主动撤销。使用 Redis 存储已失效的 Token 实现黑名单机制。

```java
// JWT 工具类扩展
@Component
public class JwtUtil {

    @Autowired
    private RedisUtil redisUtil;

    private static final long EXPIRATION_TIME = 7200_000L;  // 2小时

    // 生成 Token
    public String generateToken(String userId, String username, String role) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("userId", userId);
        claims.put("username", username);
        claims.put("role", role);

        String token = Jwts.builder()
                .claims(claims)
                .subject(userId)
                .issuedAt(new Date(System.currentTimeMillis()))
                .expiration(new Date(System.currentTimeMillis() + EXPIRATION_TIME))
                .signWith(getSigningKey())
                .compact();

        return token;
    }

    // 校验 Token 是否有效
    public boolean validateToken(String token) {
        // 1. 基本格式和签名校验
        try {
            Jwts.parser()
                .verifyWith(getSigningKey())
                .build()
                .parseSignedClaims(token);
        } catch (Exception e) {
            return false;
        }

        // 2. 检查是否在黑名单中（登出后加入黑名单）
        String blacklistKey = RedisConstants.TOKEN_BLACKLIST_KEY + token;
        if (redisUtil.hasKey(blacklistKey)) {
            return false;
        }

        return true;
    }

    // 将 Token 加入黑名单（登出时调用）
    public void invalidateToken(String token) {
        // 计算 Token 剩余有效时间
        long remainingTime = getExpirationTime(token);
        if (remainingTime <= 0) {
            return;  // Token 已过期
        }

        // 将 Token 存入黑名单，过期时间与 Token 剩余时间一致
        String blacklistKey = RedisConstants.TOKEN_BLACKLIST_KEY + token;
        redisUtil.set(blacklistKey, "1", remainingTime);
    }

    // 从请求头提取 Token
    public String extractToken(HttpServletRequest request) {
        String bearerToken = request.getHeader("Authorization");
        if (bearerToken != null && bearerToken.startsWith("Bearer ")) {
            return bearerToken.substring(7);
        }
        return null;
    }
}

// 登出接口使用示例
@PostMapping("/logout")
public JSONReturn logout(HttpServletRequest request) {
    String token = jwtUtil.extractToken(request);
    if (token != null) {
        jwtUtil.invalidateToken(token);
    }
    return JSONReturn.success("登出成功");
}
```

#### 3.5.5 缓存三大问题及其解决方案

| 问题 | 描述 | 解决方案 |
|------|------|----------|
| **缓存击穿** | 一个热点 key 过期瞬间，大量并发请求直接打到数据库 | 1. 热点数据永不过期<br>2. 使用分布式锁控制只有一个请求去查库<br>3. 缓存预热 |
| **缓存穿透** | 请求查询一个不存在的数据，缓存未命中导致每次都查库 | 1. 将空值也缓存（设置较短的 TTL）<br>2. 使用布隆过滤器（Bloom Filter）预判 key 是否存在 |
| **缓存雪崩** | 大量 key 在同一时间集中过期，或 Redis 宕机 | 1. TTL 增加随机值避免集中过期<br>2. 多级缓存架构（本地缓存 + Redis）<br>3. Redis 集群高可用 |

```java
// 缓存击穿解决方案：分布式锁保护热点数据查询
public News getHotNews(Integer id) {
    String key = RedisConstants.NEWS_KEY + id;

    // 先查缓存
    Object obj = redisUtil.get(key);
    if (obj != null) {
        return (News) obj;
    }

    // 缓存未命中，使用分布式锁控制并发
    String lockKey = RedisConstants.LOCK_KEY + "hot_news:" + id;
    String requestId = UUID.randomUUID().toString();

    try {
        // 尝试获取锁
        if (distributedLock.tryLock(lockKey, requestId, 5)) {
            // 双重检查（可能已有线程刚完成查询）
            obj = redisUtil.get(key);
            if (obj != null) {
                return (News) obj;
            }

            // 查询数据库
            News news = newsMapper.selectById(id);
            if (news != null) {
                // 热点数据设置较长过期时间
                redisUtil.set(key, news, RedisConstants.NEWS_TTL * 2);
            }
            return news;
        } else {
            // 获取锁失败，等待后重试或返回空
            Thread.sleep(100);
            return getHotNews(id);  // 简单重试，实际应限制重试次数
        }
    } catch (Exception e) {
        return newsMapper.selectById(id);
    } finally {
        distributedLock.releaseLock(lockKey, requestId);
    }
}

// 缓存穿透解决方案：空值缓存
public News getNewsById(Integer id) {
    String key = RedisConstants.NEWS_KEY + id;
    Object obj = redisUtil.get(key);

    if (obj != null) {
        // 判断是否为空值标记
        if ("NULL".equals(obj)) {
            return null;
        }
        return (News) obj;
    }

    // 查数据库
    News news = newsMapper.selectById(id);

    if (news == null) {
        // 数据不存在，缓存空值标记（较短 TTL）
        redisUtil.set(key, "NULL", 60);  // 缓存1分钟
        return null;
    }

    // 数据存在，正常缓存
    redisUtil.set(key, news, RedisConstants.NEWS_TTL);
    return news;
}

// 缓存雪崩解决方案：TTL 随机化
private long getRandomTtl(long baseTtl) {
    // 在基础 TTL 上增加 ±30% 随机值，避免集中过期
    double factor = 0.7 + Math.random() * 0.6;  // 0.7 ~ 1.3
    return (long) (baseTtl * factor);
}
```

#### 3.5.6 Redis 连接池与性能调优建议

```yaml
# 生产环境推荐配置（application.yml）
spring:
  data:
    redis:
      host: ${REDIS_HOST:127.0.0.1}
      port: ${REDIS_PORT:6379}
      password: ${REDIS_PASSWORD:}
      timeout: 3000ms

      # 使用 Lettuce 连接池（Spring Boot 3 默认）
      lettuce:
        pool:
          # 最大连接数：根据实际业务并发调整，推荐 = 业务并发峰值 / 每个请求 Redis 操作次数
          max-active: 50
          # 最大空闲连接：保持预热，减少连接创建开销
          max-idle: 20
          # 最小空闲连接
          min-idle: 5
          # 连接获取最大等待时间
          max-wait: 3000ms
        # 关闭超时连接
        shutdown-timeout: 100ms
```

**Redis 使用最佳实践：**
1. **控制 key 大小**：key 尽量简短（但要有可读性），value 避免过大
2. **合理设置过期时间**：所有缓存数据必须设置 TTL，避免内存膨胀
3. **避免大 key**：单个 key 的 value 不要超过 10KB，大对象考虑拆分
4. **批量操作**：使用 mget/mset 替代多次 get/set 减少网络往返
5. **监控 Redis**：监控内存使用率、命中率、QPS、慢查询
6. **key 命名规范**：`业务:对象:标识`，如 `news:detail:123`
